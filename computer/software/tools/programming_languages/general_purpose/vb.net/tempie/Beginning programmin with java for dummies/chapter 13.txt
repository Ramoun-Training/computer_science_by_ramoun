PrintStream class:
==================
import java.io.PrintStream;
class print {
    public static void main(String args []) {
        PrintStream fileWriter = new PrintStream("file.txt");
        fileWriter.print("this is some text.");
        fileWriter.println("\t lets end up this line ");
        fileWriter.close();
    }
}//write to files.
